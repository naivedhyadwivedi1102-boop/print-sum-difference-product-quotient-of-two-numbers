# print-sum-difference-product-quotient-of-two-numbers
#include <stdio.h>

int main(){
    float a,b;
    printf("enter a : ");
    scanf("%f",&a);
    printf("enter b : ");
    scanf("%f",&b);
    printf("sum : %f\n",a+b);
    printf("difference : %f\n",a-b);
    printf("product : %f\n",a*b);
    printf("quotient : %f\n",a/b);
    return 0;
}
